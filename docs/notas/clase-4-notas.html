Crear etiquetas (labels)

Con mutate + case_when

temp_df <- temp_df |> 
  mutate(REGION2 = 
           case_when(
             REGION == 1 ~ "AMBA",
             REGION != 1 ~ "Interior",
            .default = "No Encontrado"))
